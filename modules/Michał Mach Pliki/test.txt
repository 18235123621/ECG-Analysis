function read_file(f::IOStream)
	lines = readlines(f);
	counter = 1
	for l in lines
   		counter += 1
	end
	val = zeros(counter)
	counter=1;
	for l in lines
   		val[counter]=parse(replace(strip(l),",","."));
		counter+=1;
	end
	return val;
end
